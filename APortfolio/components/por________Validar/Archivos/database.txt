--- ASIGNACION DEL PERMISOS, TODOS LOS PERFILES
declare @id_permisoaccion as int
DECLARE @tabla TABLE (idfuncionalidad int, revisado int default 0)
INSERT INTO @tabla (idfuncionalidad)
(select id from gui_funcionalidad where active = 1 and 
(nombreControl = 'inc_contactosInternos' and funcionalidad = 'Personal Asignado Cliente' and id_funcionalidad = @id_funcionalidadPadre and id_tipoFuncionalidad = 5))
 

declare @cantidad int = (SELECT COUNT ( DISTINCT idfuncionalidad) from @tabla )
declare @RT as integer = 0
declare @id as integer = 0
while @RT < @cantidad
begin

set @id = (select top 1 idfuncionalidad from @tabla where revisado = 0)
set @id_permisoaccion =(select top 1 id from gui_funcionalidad where id = @id) 
if not exists( select * from seg_perfil_funcionalidad where id_funcionalidad = @id_permisoaccion) 

begin 
INSERT INTO [seg_perfil_funcionalidad]
([uid]
,[eid]
,[id_usuario_modifico]
,[id_usuario_creo]
,[fechaModificacion]
,[fechaCreacion]
,[active]
,[perfil_funcionalidad]
,[id_perfil]
,[id_funcionalidad])

select 
@licencia,
'01',
1,
1,
getdate(),
getdate(),
1,
'',
id,
@id_permisoaccion 
from seg_perfil 

end

UPDATE @tabla set revisado = 1 where revisado = 0 and idfuncionalidad = @id
SET @RT = @RT + 1
end
